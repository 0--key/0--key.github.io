Raw and obscure description

Test the ability to infer by submitting a poorly formatted task.

Description

Consider a sequence u where u is defined as follows:

The number u(0) = 1 is the first one in u. For each x in u, then y = 2 * x + 1 and z = 3 * x + 1 must be in u too. There are no other numbers in u. Ex: u = [1, 3, 4, 7, 9, 10, 13, 15, 19, 21, 22, 27, …]

1 gives 3 and 4, then 3 gives 7 and 10, 4 gives 9 and 13, then 7 gives 15 and 22 and so on…

Task:

Given parameter n the function dbl_linear (or dblLinear…) returns the element u(n) of the ordered (with <) sequence u.

Example:

dbl_linear(10) should return 22

Note:

Focus attention on efficiency

Answer

  from heapq import heappop, heappush

  def dbl_linear(n):
      u = [1]
      seen = {1}
    
      for _ in range(n):
          x = heappop(u)
          for y in [2 * x + 1, 3 * x + 1]:
              if y not in seen:
                  seen.add(y)
                  heappush(u, y)
                
      return heappop(u)

  # Example assertion to validate the function
  assert(dbl_linear(10) == 22)

A more complex task

Description

For a given chemical formula represented by a string, count the number of atoms of each element contained in the molecule and return an object.

For example:

water = 'H2O' parse_molecule(water) # return {H: 2, O: 1}

magnesium_hydroxide = 'Mg(OH)2' parse_molecule(magnesium_hydroxide) # return {Mg: 1, O: 2, H: 2}

var fremy_salt = 'K4[ON(SO3)2]2' parse_molecule(fremySalt) # return {K: 4, O: 14, N: 2, S: 4}

As you can see, some formulas have brackets in them. The index outside the brackets tells you that you have to multiply count of each atom inside the bracket on this index. For example, in Fe(NO3)2 you have one iron atom, two nitrogen atoms and six oxygen atoms.

Note that brackets may be round, square or curly and can also be nested. Index after the braces is optional.

Answer

import re
from collections import defaultdict

def parse_molecule(formula):
    def multiply_dict(d, n):
        return {k: v * n for k, v in d.items()}

    def parse_section(section):
        atoms = defaultdict(int)
        i = 0
        while i < len(section):
            if section[i].isalpha():
                j = i + 1
                while j < len(section) and section[j].islower():
                    j += 1
                element = section[i:j]
                k = j
                while k < len(section) and section[k].isdigit():
                    k += 1
                count = int(section[j:k] or 1)
                atoms[element] += count
                i = k
            elif section[i] in '([{':
                stack = []
                brackets = {'(': ')', '[': ']', '{': '}'}
                close_bracket = brackets[section[i]]
                for j in range(i, len(section)):
                    if section[j] == section[i]:
                        stack.append(section[i])
                    elif section[j] == close_bracket:
                        stack.pop()
                    if not stack:
                        break
                content = parse_section(section[i + 1: j])
                k = j + 1
                while k < len(section) and section[k].isdigit():
                    k += 1
                multiplier = int(section[j + 1: k] or 1)
                content = multiply_dict(content, multiplier)
                for key, val in content.items():
                    atoms[key] += val
                i = k
        return atoms

    return parse_section(formula)

Task description

You have to create a function that takes a positive integer number and returns the next bigger number formed by the same digits:

next_bigger(12)==21 next_bigger(513)==531 next_bigger(2017)==2071

If no bigger number can be composed using those digits, return -1:

next_bigger(9)==-1 next_bigger(111)==-1 next_bigger(531)==-1

Answer

def next_bigger(n):
    digits = list(str(n))
  
    # Find the rightmost digit that is smaller than the digit next to it
    for i in range(len(digits) - 2, -1, -1):
        if digits[i] < digits[i + 1]:
            break
    else:
        return -1
  
    # Find the smallest digit on the right that is larger than digits[i]
    for j in range(len(digits) - 1, i, -1):
        if digits[j] > digits[i]:
            break
  
    # Swap the two digits
    digits[i], digits[j] = digits[j], digits[i]
  
    # Reverse the digits to the right of i
    digits = digits[:i + 1] + sorted(digits[i + 1:])
  
    return int(''.join(digits))

def parse_molecule(formula):
    def multiply_dict(d, n):
        for k in d:
            d[k] *= n
        return d
    
    def parse_section(section, multiplier=1):
        atoms = {}
        stack = []
        multipliers = [multiplier]
        n = len(section)
        i = 0
        
        while i < n:
            if section[i].isalpha():
                start = i
                i += 1
                while i < n and section[i].islower():
                    i += 1
                element = section[start:i]
                start = i
                while i < n and section[i].isdigit():
                    i += 1
                count = int(section[start:i]) if start != i else 1
                atoms[element] = atoms.get(element, 0) + count * multipliers[-1]
            elif section[i] in '([{':
                stack.append((atoms, multipliers.pop()))
                atoms, multipliers[-1] = {}, 1
                multipliers.append(multiplier)
                i += 1
            elif section[i] in ')]}':
                i += 1
                start = i
                while i < n and section[i].isdigit():
                    i += 1
                sub_multiplier = int(section[start:i]) if start != i else 1
                if stack:
                    atoms = multiply_dict(atoms, sub_multiplier)
                    prev_atoms, prev_multiplier = stack.pop()
                    for k, v in atoms.items():
                        prev_atoms[k] = prev_atoms.get(k, 0) + v * prev_multiplier
                    atoms = prev_atoms
                multipliers.pop()
            else:
                i += 1

        return atoms
    
    return parse_section(formula)

Implementation

Here's the implementation of a double-linked list in Python with a basic test suite to verify its main functionalities. 

class Node:
    def __init__(self, value):
        self.value = value
        self.next = None
        self.prev = None

class DoubleLinkedList:
    def __init__(self):
        self.head = None
        self.tail = None
        
    def add_node(self, value):
        new_node = Node(value)
        if not self.head:
            self.head = self.tail = new_node
        else:
            self.tail.next = new_node
            new_node.prev = self.tail
            self.tail = new_node
            
    def delete_node(self, value):
        current = self.head
        while current:
            if current.value == value:
                if current.prev:
                    current.prev.next = current.next
                else:
                    self.head = current.next
                if current.next:
                    current.next.prev = current.prev
                else:
                    self.tail = current.prev
                return True
            current = current.next
        return False
    
    def find_node(self, value):
        current = self.head
        while current:
            if current.value == value:
                return current
            current = current.next
        return None
    
    def to_list(self):
        result = []
        current = self.head
        while current:
            result.append(current.value)
            current = current.next
        return result

Here's the updated implementation of the DoubleLinkedList class with the feature to ensure unique node values. If an attempt is made to insert a node with a non-unique value, an Exception is raised with the message "This value is already in the list!". 

class Node:
    def __init__(self, value):
        self.value = value
        self.next = None
        self.prev = None

class DoubleLinkedList:
    def __init__(self):
        self.head = None
        self.tail = None
        self.values = set()
        
    def add_node(self, value):
        if value in self.values:
            raise Exception("This value is already in the list!")
        
        new_node = Node(value)
        if not self.head:
            self.head = self.tail = new_node
        else:
            self.tail.next = new_node
            new_node.prev = self.tail
            self.tail = new_node
        self.values.add(value)
            
    def delete_node(self, value):
        current = self.head
        while current:
            if current.value == value:
                self.values.remove(value)
                if current.prev:
                    current.prev.next = current.next
                else:
                    self.head = current.next
                if current.next:
                    current.next.prev = current.prev
                else:
                    self.tail = current.prev
                return True
            current = current.next
        return False
    
    def find_node(self, value):
        current = self.head
        while current:
            if current.value == value:
                return current
            current = current.next
        return None

    def __repr__(self):
        result = []
        current = self.head
        while current:
            result.append(current.value)
            current = current.next
        return str(result)

class DoubleLinkedList:
    def __init__(self):
        self.head = None
        self.tail = None
        self.values = set()

    def add_node(self, value):
        if value in self.values:
            raise Exception("This value is already in the list!")

        new_node = Node(value)
        if not self.head:
            self.head = self.tail = new_node
        else:
            self.tail.next = new_node
            new_node.prev = self.tail
            self.tail = new_node
        self.values.add(value)

    def delete_node(self, value):
        current = self.head
        while current:
            if current.value == value:
                self.values.remove(value)
                if current.prev:
                    current.prev.next = current.next
                else:
                    self.head = current.next
                if current.next:
                    current.next.prev = current.prev
                else:
                    self.tail = current.prev
                return True
            current = current.next
        return False

    def find_node(self, value):
        current = self.head
        while current:
            if current.value == value:
                return current
            current = current.next
        return None

    def __repr__(self):
        result = []
        current = self.head
        while current:
            result.append(current.value)
            current = current.next
        return str(result)
